Memory Update Flow

// Unity fires a memory event after a player-NPC interaction
public async Task RecordMemoryEvent(NPC npc, PlayerAction action)
{
    var payload = new MemoryEventPayload {
        NpcId       = npc.Id,
        PlayerId    = player.Id,
        ActionType  = action.Type,           // e.g. BETRAYAL, KINDNESS, THREAT
        Context     = action.ContextSummary, // scene, stakes, witnesses
        Timestamp   = DateTime.UtcNow
    };

    // Send to Azure Function → GPT-4o processes emotional weight
    var updatedMemory = await AzureMemoryService.ProcessEvent(payload);

    // Cosmos DB stores updated NPC emotional state graph
    await CosmosMemoryStore.Upsert(npc.Id, updatedMemory);

    // Propagate gossip to nearby NPCs via Azure Service Bus
    if (action.WitnessCount > 0)
        await GossipBus.Broadcast(npc.SocialGraph, payload);
}

// Next time player speaks to this NPC — memory is injected live
string BuildDialogueContext(NPC npc) =>
    $"""
    You are {npc.Name}. Your emotional memory of this player:
    Trust: {npc.Memory.TrustScore}/100
    Key events: {npc.Memory.SalientEvents}
    Current mood: {npc.Memory.CurrentEmotion}
    Respond naturally. Do not break character.
    """;
